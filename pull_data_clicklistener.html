You need to click something!

What makes it work:

(function() {
    var tableid, layer, target, output, options, map, entry;

    tableid = 377240;
    layer = new google.maps.FusionTablesLayer(tableid, {
        suppressInfoWindows: true
    });
    target = document.getElementById('map_canvas');
    output = document.getElementById('output');
    options = {
        center: new google.maps.LatLng(31.25, -99.84),
        zoom: 5,
        mapTypeId: 'roadmap',
        streetViewControl: false
    };

    map = new google.maps.Map(target, options);
    layer.setMap(map);

    google.maps.event.addListener(layer, 'click', function(e) {
        entry = e.row['county_proper'].value;
        output.innerHTML = "You clicked " + entry + "!";
    });

}).call(this);
